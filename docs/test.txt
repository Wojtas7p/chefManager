<script>
const API = 'http://localhost:3000/api';
let token = null;
let currentSupplierId = null;

// ================== REJESTRACJA ==================
async function register() {
  const login = document.getElementById('regLogin').value;
  const password = document.getElementById('regPassword').value;

  await fetch(`${API}/auth/register`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ login, password })
  });
  alert('Zarejestrowano! Teraz się zaloguj.');
}

// ================== LOGOWANIE ==================
async function login() {
  const loginVal = document.getElementById('loginLogin').value;
  const password = document.getElementById('loginPassword').value;

  const res = await fetch(`${API}/auth/login`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ login: loginVal, password })
  });

  const data = await res.json();
  if (data.token) {
    token = data.token;
    alert('Zalogowano!');
    loadSuppliers();
  } else {
    alert('Błąd logowania');
  }
}

// ================== DOSTAWCY ==================
async function loadSuppliers() {
  const res = await fetch(`${API}/suppliers`, {
    headers: { 'Authorization': `Bearer ${token}` }
  });
  const suppliers = await res.json();

  const ul = document.getElementById('suppliers');
  ul.innerHTML = '';

  suppliers.forEach(s => {
    const li = document.createElement('li');
    li.textContent = s.name;
    li.onclick = () => selectSupplier(s._id);
    ul.appendChild(li);
  });
}

async function addSupplier() {
  const name = document.getElementById('supplierName').value;
  await fetch(`${API}/suppliers`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`
    },
    body: JSON.stringify({ name })
  });
  loadSuppliers();
}

function selectSupplier(id) {
  currentSupplierId = id;
  loadProducts();
}

// ================== PRODUKTY ==================
async function loadProducts() {
  const res = await fetch(`${API}/products/${currentSupplierId}`, {
    headers: { 'Authorization': `Bearer ${token}` }
  });
  const products = await res.json();

  const ul = document.getElementById('products');
  ul.innerHTML = '';

  products.forEach(p => {
    const li = document.createElement('li');
    li.textContent = p.name;
    ul.appendChild(li);
  });
}

async function addProduct() {
  if (!currentSupplierId) return alert('Wybierz dostawcę');
  const name = document.getElementById('productName').value;

  await fetch(`${API}/products`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`
    },
    body: JSON.stringify({ name, supplierId: currentSupplierId })
  });
  loadProducts();
}

const User = require('../models/User');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');

exports.register = async (req, res) => {
  try {
    // sprawdzamy, czy login już istnieje
    const exists = await User.findOne({ login: req.body.login });
    if (exists) return res.status(400).json({ error: 'Login już istnieje' });

    const hashed = await bcrypt.hash(req.body.password, 10);

    // tworzymy admina, owner = sam siebie
    const user = new User({
      login: req.body.login,
      password: hashed,
      role: 'ADMIN',
      owner: null, // tymczasowo null, ustawimy po zapisie
      permissions: { canAddSuppliers: true, canAddProducts: true, readOnly: false }
    });

    await user.save();

    // po zapisaniu ustawiamy owner = własne _id
    user.owner = user._id;
    await user.save();

    res.status(201).json({ message: 'Admin zarejestrowany' });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
};

exports.login = async (req, res) => {
  try {
    const user = await User.findOne({ login: req.body.login });
    if (!user) return res.status(401).json({ error: 'Błąd logowania' });

    const ok = await bcrypt.compare(req.body.password, user.password);
    if (!ok) return res.status(401).json({ error: 'Błąd logowania' });

    const token = jwt.sign(
      {
        userId: user._id,
        owner: user.owner,
        role: user.role,
        permissions: user.permissions
      },
      process.env.JWT_SECRET,
      { expiresIn: '1d' }
    );

    res.json({ token });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
};

const Message = require('../models/Message');

/**
 * get message
 */
exports.getMessages = async (req, res) => {
  const { receiverId } = req.query;

  const filter = {
    owner: req.ownerId
  };

  if (receiverId) {
    filter.$or = [
      { sender: req.userId, receiver: receiverId },
      { sender: receiverId, receiver: req.userId }
    ];
  } else {
    filter.receiver = null; 
  }

  const messages = await Message.find(filter)
    .populate('sender', 'name')
    .sort({ createdAt: 1 });

  res.json(messages);
};

/**
 * Send message
 */
exports.sendMessage = async (req, res) => {
  const { text, receiverId } = req.body;

  const message = new Message({
    owner: req.ownerId,
    sender: req.userId,
    receiver: receiverId || null,
    text
  });

  await message.save();
  res.status(201).json(message);
};

const Schedule = require('../models/Schedule');
const TimeOff = require('../models/TimeOff');


exports.saveSchedule = async (req, res) => {
  const { month, users } = req.body;

  const schedule = await Schedule.findOneAndUpdate(
    { owner: req.ownerId, month },
    { users },
    { upsert: true, new: true }
  ).populate('users.user', 'name');

  res.json(schedule);
};


exports.getSchedule = async (req, res) => {
  const { month } = req.query;

  const schedule = await Schedule.findOne({
    owner: req.ownerId,
    month
  }).populate('users.user', 'name');

  if (!schedule) return res.json(null);

  const timeOffs = await TimeOff.find({
    owner: req.ownerId
  });

  res.json({ schedule, timeOffs });
};

middleware/auth.js
-----------------------------------------------

const jwt = require('jsonwebtoken');

module.exports = (req, res, next) => {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'Brak tokenu' });

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);

    req.userId = decoded.userId;
    req.ownerId = decoded.owner;       // 🔥 KLUCZ
    req.userRole = decoded.role;
    req.userPermissions = decoded.permissions;

    next();
  } catch {
    res.status(401).json({ error: 'Nieprawidłowy token' });
  }
};
